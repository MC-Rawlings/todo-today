/**
 * A list gets created which will hold tasks
 * @param {string} title Title that gets displayed for the list
 * @returns functions
 * @example
 * createList('New title');
 * // Creates a new list with a tile of 'New title'
 */

const myList = createList('Hello');

const createList = (title) => {
  let list = [];

  // Getters

  /**
   * @method module:ns/SDK/Data.Foo#bar
   * @param {string} str
   * @return {Promise.<bool>}
   * @example
   * var foo = new Foo();
   * foo.bar('hello world').then(function (val) {});
   */
  const getTitle = () => title;
  const getList = () => list;

  // Setters
  // eslint-disable-next-line no-param-reassign
  // eslint-disable-next-line no-return-assign
  const setTitle = (newTitle) => (title = newTitle);

  // eslint-disable-next-line no-return-assign
  const updateList = (newList) => (list = newList);

  // Methods
  /**
   * Gets a title
   * @returns {string} The title
   */
  const addToList = (task) => list.push(task);

  const removeTask = (index) => {
    list.splice(index, 1);
  };

  return {
    getTitle,
    getList,
    setTitle,
    addToList,
    removeTask,
    updateList,
  };
};

export default createList;
